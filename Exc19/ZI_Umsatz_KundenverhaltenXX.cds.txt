@AbapCatalog.viewEnhancementCategory: [#NONE]
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'UmsatzKundenverhalten'
@Metadata.ignorePropagatedAnnotations: false
@AbapCatalog.entityBuffer.definitionAllowed
@Metadata.allowExtensions: true
define view entity ZI_Umsatz_KundenverhaltenXX
  as select from ZI_BuchungXX         as Buchung
    inner join   ZI_Umsatz_pro_FlugXX as Umsatz on  Buchung.Carrid = Umsatz.Carrid
                                                and Buchung.Connid = Umsatz.Connid
{
  @EndUserText.label: 'GesellschaftId'
  key Buchung.Carrid,
  @EndUserText.label: 'VerbindungsId'
  key Buchung.Connid,
  @Semantics.amount.currencyCode: 'Currency_EUR'
  @EndUserText.label: 'Umsatz_Flugverbindung'
  key Umsatz.Umsatz_pro_Flugverbindung,
  @EndUserText.label: 'Waehrung'
  Umsatz.Currency_EUR,
  @EndUserText.label: 'Durchschn. Buchung Voraus'
  avg(Buchung.Buchung_im_Voraus as abap.dec(12,2)) as Durchschn_Buchung_Voraus
}
group by
  Buchung.Carrid,
  Buchung.Connid,
  Umsatz.Umsatz_pro_Flugverbindung,
  Umsatz.Currency_EUR
