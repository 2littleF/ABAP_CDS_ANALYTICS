@AbapCatalog.viewEnhancementCategory: [#NONE]
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'BuchungRabatt'
@Metadata.ignorePropagatedAnnotations: true
define view entity ZI_Buchung_RabattXX as select from ZI_BuchungXX as Buchung
inner join ZI_KundeXX as Kunde
    on Buchung.Customid = Kunde.Id 
inner join ZI_FlugXX as Flug
    on Buchung.Carrid = Flug.Carrid
    and Buchung.Connid = Flug.Connid
{
    Buchung.Carrid,
    Buchung.Connid,
    Buchung.Fldate,
    Buchung.Bookid,
    Buchung.Customid,
    Kunde.Custtype,
    @Semantics.amount.currencyCode: 'Currency_EUR'
    Flug.Preis_in_Euro,
    Flug.Currency_EUR,
    @Semantics.amount.currencyCode: 'Currency_EUR'    
    cast( (cast(Flug.Preis_in_Euro as abap.fltp) * 0.9) as s_price) as Preis_Eur_Rabatt
} where Kunde.Custtype = 'B'
union
select from ZI_BuchungXX as Buchung
inner join ZI_KundeXX as Kunde
    on Buchung.Customid = Kunde.Id 
inner join ZI_FlugXX as Flug
    on Buchung.Carrid = Flug.Carrid
    and Buchung.Connid = Flug.Connid
{
    Buchung.Carrid,
    Buchung.Connid,
    Buchung.Fldate,
    Buchung.Bookid,
    Buchung.Customid,
    Kunde.Custtype, 
    Flug.Preis_in_Euro,
    Flug.Currency_EUR,       
    cast( (cast(Flug.Preis_in_Euro as abap.fltp) * 1) as s_price) as Preis_Eur_Rabatt
} where Kunde.Custtype = 'P'
